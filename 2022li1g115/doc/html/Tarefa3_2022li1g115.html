A função "moveJogador1" consoante o mapa fornecido e a posiçao de um jogador executa o movimento dado 

moveJogador1 (Mapa l ((Rio v,os):t)) (Jogador (x,y)) (Parado) 
    | Tronco == ((!!) (snd ((!!) ((Rio v,os):t) y)) x) = (Jogador ((x + v),y))
    | otherwise = (Jogador (x,y))
moveJogador1 (Mapa l ((_,os):t)) (Jogador (x,y)) (Parado) = (Jogador (x,y))                                                            
moveJogador1 (Mapa l ls) (Jogador (x,y)) (Move Baixo) 
    | ((length ls) - 1 == y) || ((y + 1) < (length ls) && Arvore == ((!!) (snd ((!!) ls (y+1))) x)) = (Jogador (x,y))  
    | otherwise = (Jogador (x,(y + 1)))
moveJogador1 (Mapa l ls) (Jogador (x,y)) (Move Cima) 
    | y == 0 || Arvore == ((!!) (snd ((!!) ls (y-1))) x) = (Jogador (x,y))
    | otherwise = (Jogador (x,(y-1)))
moveJogador1 (Mapa l ((_,os):t)) (Jogador (x,y)) (Move Esquerda) 
    | x == 0 || Arvore == ((!!) os (x-1)) = (Jogador (x,y))
    | otherwise = (Jogador ((x-1),y))
moveJogador1 (Mapa l ((_,os):t)) (Jogador (x,y)) (Move Direita) 
    | x == (l-1) || Arvore == ((!!) os (x+1)) = (Jogador (x,y))
    | otherwise = (Jogador ((x+1),y))
